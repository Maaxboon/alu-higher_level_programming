#!/usr/bin/python3
MyList = __import__('1-my_list').MyList

def test_my_list():
    # Create an instance of MyList
    my_list = MyList()
    
    # Add elements to the list
    my_list.append(1)
    my_list.append(4)
    my_list.append(2)
    my_list.append(3)
    my_list.append(5)
    
    # Print the original list
    print("Original list:", my_list)  # Output: [1, 4, 2, 3, 5]
    
    # Print sorted list
    print("Sorted list:", end=" ")
    my_list.print_sorted()  # Output: [1, 2, 3, 4, 5]
    
    # Check if the original list remains unchanged
    print("Original list after sorting:", my_list)  # Output: [1, 4, 2, 3, 5]

if __name__ == "__main__":
    test_my_list()
